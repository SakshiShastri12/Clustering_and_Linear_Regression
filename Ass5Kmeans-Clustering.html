# Calculate the purity for each cluster
purity_percentage = []
for i in range(optimal_k):
    total = conf_matrix[:, i].sum()
    majority_class = max(conf_matrix[:, i])
    purity_cluster = (majority_class / total) * 100
    purity_percentage.append(purity_cluster)

print("Purity of clusters in percentage:", ", ".join([f"{p:.2f}%" for p in purity_percentage]))
